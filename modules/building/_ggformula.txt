Example using formula

You can see a step through by removing some code.

install.packages("ggformula")

data(penguins, package = "palmerpenguins")
library(ggformula)
penguins %>% 
  dplyr::filter(., sex %in% c("male", "female")) |>

  ggformula::set_variable_labels(
    bill_length_mm = "bill length (mm)",
    bill_depth_mm = "bill depth (mm)"
  ) %>%
  ggformula::gf_jitter(bill_length_mm ~ bill_depth_mm | island ~ sex,
                       color = ~ species,
                       width = 0.05, 
                       height = 0.05, 
                       size = 0.8, 
                       alpha = 0.6) %>%
  ggformula::gf_density2d(alpha = 0.5) %>%
  ggformula::gf_labs(
    title = "Palmer Penguins",
    caption = "Data available in palmerpenguins package"
  ) %>%
  ggformula::gf_refine(scale_color_brewer(type = "qual")) %>%
  ggformula::gf_theme(theme_bw()) %>%
  ggformula::gf_theme(
    legend.position = 'top',
    text = element_text(colour = "navy", face = "italic")
  )
